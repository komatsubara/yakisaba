# ここに使いたいプラグインを書いておく
# zplug "ユーザー名/リポジトリ名", タグ

# 自分自身をプラグインとして管理
zplug "zplug/zplug", hook-build:'zplug --self-manage'

# インストールしてないプラグインはインストール
if ! zplug check --verbose; then
	    printf "Install? [y/N]: "
			    if read -q; then
						        echo; zplug install
										    fi
												fi

# コマンドをリンクして、PATH に追加し、プラグインは読み込む
